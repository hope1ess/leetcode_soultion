# 198.打家劫舍
递归解  

```
var rob = function(nums) {
   return fn(nums.length, nums);
};

function fn(n, nums) {
    if (n === 0) return 0;
    if (n === 1) return nums[0];
    return Math.max(fn(n - 1, nums), fn(n - 2, nums) + nums[n - 1])
}
```
状态转移方程

```
var rob = function(nums) {
    let prere = 0, pre = nums[0];
    for (let i = 1; i < nums.length; ++i) {
        const temp = pre;
        pre = Math.max(pre, prere + nums[i]);
        prere = temp;
    }
    return pre;
};
```
